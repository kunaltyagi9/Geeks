class GFG {
	public static void main (String[] args) {
		Scanner sc = new Scanner(System.in);
		int t = sc.nextInt();
		for(int i = 0 ; i < t ; i++){
			int n = sc.nextInt();
			int arr[] = new int[n];
			int sum = 0,s = 0;
			for(int j = 0 ; j < n-1 ; j++){
				arr[j] = sc.nextInt();
				sum+=arr[j];
			}
			while(n>0){
				s+=n;
				n--;
			}
			System.out.println(s-sum);
		}
	}
}
